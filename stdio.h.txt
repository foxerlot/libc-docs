`#include <stdio.h>`

This header provides functionality for input and output operations, including file handling and formatted I/O.

-# MACROS

#define EOF /* implementation-defined */
	Negative integer constant returned on end-of-file or error.

#define FILENAME_MAX /* implementation-defined */
	Maximum supported filename length.

#define FOPEN_MAX /* implementation-defined */
	Maximum number of files that can be open simultaneously.

#define BUFSIZ /* implementation-defined */
	Size of the buffer used by setbuf.

#define SEEK_SET /* implementation-defined */
	Seek relative to beginning of file.

#define SEEK_CUR /* implementation-defined */
	Seek relative to current file position.

#define SEEK_END /* implementation-defined */
	Seek relative to end of file.

#define TMP_MAX /* implementation-defined */
	Maximum number of unique names generated by tmpnam.

#define L_tmpnam /* implementation-defined */
	Size of character array required by tmpnam.

-# TYPES

typedef /* implementation-defined */ FILE;
	Object type capable of holding information about a file stream.

typedef /* implementation-defined */ fpos_t;
	Object type capable of holding file position information.

typedef /* implementation-defined */ size_t;
	Unsigned integer type returned by sizeof.

-# STREAMS

extern FILE *stdin;
	Standard input stream.

extern FILE *stdout;
	Standard output stream.

extern FILE *stderr;
	Standard error output stream.

-# FUNCTIONS

-## FORMATTED INPUT/OUTPUT

int printf( const char* format, ... );
	Prints formatted output to stdout.
	format - format control string

int fprintf( FILE* stream, const char* format, ... );
	Prints formatted output to a stream.
	stream - output stream
	format - format control string

int sprintf( char* str, const char* format, ... );
	Prints formatted output to a string.
	str    - pointer to destination buffer
	format - format control string

int snprintf( char* str, size_t size, const char* format, ... );
	Prints formatted output to a string with size limit.
	str    - pointer to destination buffer
	size   - maximum number of characters to write
	format - format control string

int scanf( const char* format, ... );
	Reads formatted input from stdin.
	format - format control string

int fscanf( FILE* stream, const char* format, ... );
	Reads formatted input from a stream.
	stream - input stream
	format - format control string

int sscanf( const char* str, const char* format, ... );
	Reads formatted input from a string.
	str    - input string
	format - format control string

-## VARIADIC FORMATTED OUTPUT

int vprintf( const char* format, va_list args );
	Prints formatted output to stdout using a variable argument list.
	format - format control string
	args   - variable argument list initialized with va_start

int vfprintf( FILE* stream, const char* format, va_list args );
	Prints formatted output to a stream using a variable argument list.
	stream - output stream
	format - format control string
	args   - variable argument list

int vsprintf( char* str, const char* format, va_list args );
	Prints formatted output to a string using a variable argument list.
	str    - pointer to destination buffer
	format - format control string
	args   - variable argument list

int vsnprintf( char* str, size_t size, const char* format, va_list args );
	Prints formatted output to a string with size limit using a variable argument list.
	str    - pointer to destination buffer
	size   - maximum number of characters to write
	format - format control string
	args   - variable argument list

-## CHARACTER INPUT/OUTPUT

int getchar( void );
	Reads the next character from stdin.

int putchar( int ch );
	Writes a character to stdout.
	ch - character to write

int fgetc( FILE* stream );
	Reads the next character from a stream.
	stream - input stream

int fputc( int ch, FILE* stream );
	Writes a character to a stream.
	ch     - character to write
	stream - output stream

char* fgets( char* str, int count, FILE* stream );
	Reads a string from a stream.
	str    - pointer to destination buffer
	count  - maximum number of characters to read
	stream - input stream

int fputs( const char* str, FILE* stream );
	Writes a string to a stream.
	str    - null-terminated string to write
	stream - output stream

int puts( const char* str );
	Writes a string to stdout followed by a newline.
	str - null-terminated string to write

-## DIRECT INPUT/OUTPUT

size_t fread( void* ptr, size_t size, size_t count, FILE* stream );
	Reads binary data from a stream.
	ptr    - pointer to buffer
	size   - size of each element
	count  - number of elements to read
	stream - input stream

size_t fwrite( const void* ptr, size_t size, size_t count, FILE* stream );
	Writes binary data to a stream.
	ptr    - pointer to data to write
	size   - size of each element
	count  - number of elements to write
	stream - output stream

-## FILE MANAGEMENT

FILE* fopen( const char* filename, const char* mode );
	Opens a file.
	filename - name of the file
	mode     - file access mode

FILE* freopen( const char* filename, const char* mode, FILE* stream );
	Reopens a file with a different mode.
	filename - name of the file
	mode     - file access mode
	stream   - existing stream to reopen

int fclose( FILE* stream );
	Closes a file.
	stream - file stream to close

int remove( const char* filename );
	Deletes a file.
	filename - name of the file

int rename( const char* oldname, const char* newname );
	Renames a file.
	oldname - current filename
	newname - new filename

FILE* tmpfile( void );
	Creates a temporary file.

char* tmpnam( char* str );
	Generates a temporary filename.
	str - pointer to character array

-## FILE POSITIONING

int fseek( FILE* stream, long offset, int origin );
	Moves the file position indicator.
	stream - file stream
	offset - number of bytes to offset
	origin - SEEK_SET, SEEK_CUR, or SEEK_END

long ftell( FILE* stream );
	Returns the current file position.
	stream - file stream

void rewind( FILE* stream );
	Sets the file position indicator to the beginning.
	stream - file stream

int fgetpos( FILE* stream, fpos_t* pos );
	Gets the current file position.
	stream - file stream
	pos    - pointer to position object

int fsetpos( FILE* stream, const fpos_t* pos );
	Sets the file position.
	stream - file stream
	pos    - pointer to position object

-## ERROR HANDLING

void clearerr( FILE* stream );
	Clears error and EOF indicators.
	stream - file stream

int feof( FILE* stream );
	Checks if end-of-file indicator is set.
	stream - file stream

int ferror( FILE* stream );
	Checks if error indicator is set.
	stream - file stream

void perror( const char* str );
	Prints a descriptive error message to stderr.
	str - custom message prefix
